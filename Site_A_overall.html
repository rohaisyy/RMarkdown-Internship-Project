Imputation by Spline Interpolation

Site_A <- Site_A %>%
  mutate(SplImp_Var.1=na_interpolation(Var.1, option = "spline"), 
         SplImp_Var.2=na_interpolation(Var.2, option = "spline"), 
         SplImp_Var.3=na_interpolation(Var.3, option = "spline"))

Checking Outlier for Site A

Filtering Outliers

Site_A <- Site_A %>%
  mutate(is_outlier1 = SplImp_Var.1 > 1000, 
         is_outlier2 = SplImp_Var.2 > 10000, 
         is_outlier3 = SplImp_Var.3 > 1000) %>%
  filter(!is_outlier1, !is_outlier2, !is_outlier3)

Site A Monthly Mean

Monthly_Mean <- Site_A %>%
  group_by(Site, Year, Month) %>%
  summarize(monthly_mean_Var.1 = mean(SplImp_Var.1), 
            monthly_mean_Var.2 = mean(SplImp_Var.2), 
            monthly_mean_Var.3 = mean(SplImp_Var.3)) %>%
  arrange(Site)
